Overview

Класс предназначен для получения приложением доступа к пользовательской учетной записи и осуществления запросов к API сервиса методами GET/POST.

- (void)applicationDidFinishLaunching:(UIApplication *)application
{
    [[VKConnector sharedInstance] startWithAppID:@"YOUR_APP_ID"
                                      permissons:@[@"friends",@"wall"]];
}
